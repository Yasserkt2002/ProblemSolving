			return;
		}

		{
				GameInfo.TheCorrectAnswer=GameInfo.TowOperation.FirstNumber/GameInfo.TowOperation.SecondNumber;

					{
						GameInfo.NumberOfRightAnswer++;
						GameInfo.IsCorrectAnswer=true;
						return;
					}
				return ;
		}

		else if(GameInfo.TypeOperationType_Pragmatic==enOperationType::Mult)
		{
				GameInfo.TheCorrectAnswer=GameInfo.TowOperation.FirstNumber*GameInfo.TowOperation.SecondNumber;
				if(GameInfo.UserAnswer==(GameInfo.TowOperation.FirstNumber*GameInfo.TowOperation.SecondNumber))
					{
						GameInfo.NumberOfRightAnswer++;
						GameInfo.IsCorrectAnswer=true;
						return;
					}
				return ;
		}
				else if(GameInfo.TypeOperationType_Pragmatic==enOperationType::Sub)
				{
					GameInfo.TheCorrectAnswer=GameInfo.TowOperation.FirstNumber-GameInfo.TowOperation.SecondNumber;
					if(GameInfo.UserAnswer==(GameInfo.TowOperation.FirstNumber-GameInfo.TowOperation.SecondNumber))
					{
						GameInfo.NumberOfRightAnswer++;
						GameInfo.IsCorrectAnswer=true;
						return;
					}
					return ;
				}
		}
	
char GetOperationTypeAsChar(enOperationType OperationType)
{
	if(OperationType==enOperationType::Add)
		return '+';
	else if(OperationType==enOperationType::Div)
		return '/';
	else if(OperationType==enOperationType::Mult)
		return '*';
	else if(OperationType==enOperationType::Sub)
		return '-';
}
